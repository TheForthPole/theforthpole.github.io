        if(n%d > 0)return -1;
        ++k, a2 *= a/d, n /= d, p /= d;
        if(a2 == n)return k;
    }
    n = n*inv(a2, p), ans = bsgs(a, n, p);
    return ans == -1 ? -1 : ans+k;
}
