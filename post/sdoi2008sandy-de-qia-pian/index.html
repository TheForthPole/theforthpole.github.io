[SDOI2008]Sandy的卡片

2021-03-22

给定 nnn 个串，第 iii 个串的长度为 mim_imi​ 。字符集大小为 186418641864 。定义两个串是相同的当且仅当给某个串的每个位置都加上某个值后能成为另一个串。现在要求最大的 kkk 满足所有串都有一个相同的长度为 kkk 的子串。 40≤n≤1000,mi≤10140 \leq n\leq 1000, m_i \leq 10140≤n≤1000,mi​≤101。

大水题啊。。。
首先给每个串都做个差分，就转化为求最长公共子串了。
注意到数据范围不大。直接枚举子串在第一个串出现位置的左端点，把从这里开始的后缀拿去与其他串做 KMP\text{KMP}KMP 即可求出以这个位置开头的最长的满足条件的子串长度。把所有位置的这个值求个 max⁡\maxmax 即可。复杂度 O(nm2)\mathcal{O} (nm^2)O(nm2) 。

#include<bits/stdc++.h>
#define rg register
#define il inline
#define cn const
#define gc getchar()
#define fp(i,a,b) for(rg int i=(a),ed=(b);i<=ed;++i)
#define fb(i,a,b) for(rg int i=(a),ed=(b);i>=ed;--i)
using namespace std;
typedef cn int cint;
typedef long long LL;
il void rd(int &x){
    x = 0;
	rg int f(1); rg char c(gc);
	while(c<'0'||'9'<c){if(c=='-')f=-1;c=gc;}
	while('0'<=c&&c<='9')x=(x<<1)+(x<<3)+(c^48),c=gc;
	x*=f;
}
cint maxn = 1010, maxm = 110;
int n, len[maxn], s[maxn][maxm], nxt[maxm], ans;
il void getnxt(int *s, cint &len){
	nxt[0] = -1, nxt[1] = 0;
	for(rg int i = 2, j = 0; i <= len; ++i){
		while(~j && s[i] != s[j+1])j = nxt[j];
		nxt[i] = ++j;
	}
}
il int kmp(int *s, cint &m, int *t, cint &n){
	rg int res = 0;
	for(rg int i = 1, j = 0; i <= n; ++i){
		while(~j && t[i] != s[j+1])j = nxt[j];
		res = max(res, ++j);
	}
	return res;
}
int main(){
	rd(n), memset(s, -1, sizeof s);
	fp(i, 1, n){
		rd(len[i]), --len[i];
		fp(j, 0, len[i])rd(s[i][j]);
		fb(j, len[i], 1)s[i][j] -= s[i][j-1];
	}
	
	fp(l, 0, len[1]-1){
		getnxt(s[1]+l, len[1]-l);
		rg int res = len[1];
		fp(i, 2, n)res = min(res, kmp(s[1]+l, len[1]-l, s[i], len[i]));
		ans = max(ans, res);
	}
	printf("%d\n", ans+1);
	return 0;
}
