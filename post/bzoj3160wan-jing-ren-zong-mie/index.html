[BZOJ3160]万径人踪灭

2021-03-22

给定一个只含 a,ba,ba,b 的长度为 nnn 的字符串，需要求出有多少个子序列满足：下标和字符都关于某条对称轴对称。n≤105n \leq 10^5n≤105。

考虑暴力怎么做？暴力是不是对于每个对称轴，计算出有多少对位置关于它对称，假设有 xxx 个，再求出以它为中心的最长回文串长度，假设长度为 yyy ，那么以它为对称轴的答案就等于 2x−1−⌈y2⌉2^x - 1 - \lceil \frac{y}{2} \rceil2x−1−⌈2y​⌉ ，因为要减去连续的情况和空序列的情况。 yyy 可以 manachermanachermanacher 求，而暴力求 xxx 是 O(n2)\mathcal{O} (n^2)O(n2) 的。下面考虑优化。
如果仅仅是求有多少对下标关于对称轴对称（不考虑字符是否相等）是很好求的，直接看左右分别有多少个数即可。那么可以尝试把求 xxx 转化为求“有多少对下标对称”减去“有多少对下标对称但字符不同”。注意到字符串只含 a,ba,ba,b ，考虑这样一个函数： S(i)=[si=b]S(i)=[s_i=b]S(i)=[si​=b] ，那么容易构造这样一个运算：

S⋆S=∑k=02n−2∑0≤i,j≤n−1,i+j=k(S(i)−S(j))2=∑k=02n−2∑0≤i,j≤n−1,i+j=kS2(i)+S2(j)−2S(i)S(j)=(∑k=02n−2∑i=0min⁡(k,n−1)2S2(i))−(2∑k=02n−2∑0≤i,j≤n−1,i+j=kS(i)S(j))\begin{aligned} S \star S = \sum_{k=0}^{2n-2}\sum_{0 \leq i,j \leq n-1,i+j=k}(S(i)-S(j))^2 \\=\sum_{k=0}^{2n-2}\sum_{0 \leq i,j \leq n-1,i+j=k}S^2(i)+S^2(j)-2S(i)S(j) \\=(\sum_{k=0}^{2n-2}\sum_{i=0}^{\min(k,n-1)}2S^2(i))-(2\sum_{k=0}^{2n-2}\sum_{0 \leq i,j \leq n-1,i+j=k}S(i)S(j)) \end{aligned} S⋆S=k=0∑2n−2​0≤i,j≤n−1,i+j=k∑​(S(i)−S(j))2=k=0∑2n−2​0≤i,j≤n−1,i+j=k∑​S2(i)+S2(j)−2S(i)S(j)=(k=0∑2n−2​i=0∑min(k,n−1)​2S2(i))−(2k=0∑2n−2​0≤i,j≤n−1,i+j=k∑​S(i)S(j))​

发现这个运算刚好能求出 xxx ， 每个 kkk 后面的和式就代表对称轴为 k2\frac{k}{2}2k​ （实数运算）时下标对称但字符不同的对数。后面就是一个卷积的式子，前面是前缀和。再处理一下算重的情况，因为同一对位置会被算两次，加亿点点细节就好了。现在复杂度就是O(nlog⁡n)\mathcal{O} (n\log n)O(nlogn)了。

#include<bits/stdc++.h>
#define rg register
#define il inline
#define cn const
#define fp(i,a,b) for(rg int i=(a),ed=(b);i<=ed;++i)
#define fb(i,a,b) for(rg int i=(a),ed=(b);i>=ed;--i)
using namespace std;
typedef cn int cint;
typedef long long LL;
cint maxn = 100010, mod = 998244353, mod2 = 1e9+7, G = 3, invG = (mod+1)/3;
int n, s[maxn], f[maxn<<2], len[maxn<<1], ans;
int lim = 1, hst, rev[maxn<<2];
char ch[maxn], ch2[maxn<<1];
il int fpow(int a, int b, cint &mod, int ans = 1){
	for(; b; b >>= 1, a = 1ll*a*a%mod)if(b&1)ans = 1ll*ans*a%mod;
	return ans;
}
il void ntt(int *a, cint &typ){
	fp(i, 1, lim-1)if(rev[i] > i)swap(a[i], a[rev[i]]);
	for(rg int md = 1; md < lim; md <<= 1){
		rg int len = md<<1, Gn  =fpow(typ ? invG : G, (mod-1)/len, mod);
		for(rg int l = 0; l < lim; l += len){
			for(rg int nw = 0, Pow = 1; nw < md; ++nw, Pow = 1ll*Pow*Gn%mod){
				rg int x = a[l+nw], y = 1ll*a[l+nw+md]*Pow%mod;
				a[l+nw] = (x+y)%mod, a[l+nw+md] = (x-y+mod)%mod;
			}
		}
	}
	if(!typ)return;
	rg int inv = fpow(lim, mod-2, mod);
	fp(i, 0, lim-1)f[i] = 1ll*f[i]*inv%mod;
}
il void manacher(){
	fp(i, 1, n)ch2[i*2] = ch[i-1], ch2[i*2-1] = '#';
	ch2[0] = '@', ch2[n*2+1] = '#', ch2[n*2+2] = '$';
	rg int nw = 0, pos = 0;
	fp(i, 1, n*2+1){
		if(nw > i)len[i] = min(nw-i, len[2*pos-i]);
		else len[i] = 1;
		while(ch2[i+len[i]] == ch2[i-len[i]])++len[i];
		if(i+len[i] > nw)nw = i+len[i], pos = i;
	}
}
int main(){
	scanf("%s", ch), n = strlen(ch);
	fp(i, 0, n-1)f[i] = ch[i]-'a', s[i] = s[i-1]+f[i];
	while(lim < 2*n-1)lim <<= 1, ++hst;
	fp(i, 1, lim-1)rev[i] = (rev[i>>1]>>1)|((i&1)<<hst-1);
	ntt(f, 0);
	fp(i, 0, lim-1)f[i] = 1ll*f[i]*f[i]%mod;
	ntt(f, 1);
	fp(i, 0, 2*n-2)f[i] = s[min(n-1, i)]-f[i];
	fp(i, n, 2*n-2)f[i] -= s[i-n];
	fp(i, 0, 2*n-2){
		if(i < n)f[i] = (i+2)/2-f[i];
		else f[i] = (2*n-i)/2-f[i];
	}
	fp(i, 0, 2*n-2)f[i] = (fpow(2, f[i], mod2)+mod2-1)%mod2;
	
	manacher();
	fp(i, 0, 2*n-2)f[i] = (f[i]-len[i+2]/2+mod2)%mod2;
	fp(i, 0, 2*n-2)ans = (ans+f[i])%mod2;
	printf("%d\n", ans);
	return 0;
}
