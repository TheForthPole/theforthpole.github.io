[HEOI2015]最短不公共子串

2021-03-19

给两个小写字母串 a,ba, ba,b，请你计算：
1.1.1. aaa 的一个最短的子串，它不是 bbb 的子串。
2.2.2. aaa 的一个最短的子串，它不是 bbb 的子序列。
3.3.3. aaa 的一个最短的子序列，它不是 bbb 的子串。
4.4.4. aaa 的一个最短的子序列，它不是 bbb 的子序列。
串长不超过 200020002000。

对于前两问，我们可以先枚举 aaa 的子串的开头位置，接下来在 bbb 的后缀自动机/序列自动机上沿着转移边走，走不动时所走的长度就是以这个位置开头能匹配的 bbb 的最长子串。
对于后两问，注意到自动机的性质：如果当前确定了在哪个状态，那么剩下的能走的步数是与前面如何到达这个状态无关的。因此贪心地走最短路到达这个状态即可。因此考虑广搜，设 f[i][j]f[i][j]f[i][j]表示到达 aaa 的自动机的状态 iii 并且到达 bbb 的自动机的状态 jjj 的最少步数。每次枚举 iii 的转移边，如果 jjj 也有这条转移，则放入队列中；否则就与 bbb 失配，直接输出当前的 f+1f+1f+1 即可。

#include<bits/stdc++.h>
#define rg register
#define il inline
#define cn const
#define gc getchar()
#define fp(i,a,b) for(rg int i=(a),ed=(b);i<=ed;++i)
#define fb(i,a,b) for(rg int i=(a),ed=(b);i>=ed;--i)
#define go(u) for(rg int i=head[u];~i;i=e[i].nxt)
using namespace std;
typedef cn int cint;
typedef long long LL;
il void rd(int &x){
    x = 0;
	rg int f(1); rg char c(gc);
	while(c<'0'||'9'<c){if(c=='-')f=-1;c=gc;}
	while('0'<=c&&c<='9')x=(x<<1)+(x<<3)+(c^48),c=gc;
	x*=f;
}
cint maxn = 2010, inf = 0x3f3f3f3f;
int n, m, ans, f[maxn<<1][maxn];
char a[maxn], b[maxn];

struct SAM{
	int ch[maxn<<1][26], len[maxn<<1], fa[maxn<<1], lst, siz;
	SAM(){lst = siz = 1;}
	il void ins(cint &c, int p){
		len[++siz] = p, p = lst, lst = siz;
		for(; p && !ch[p][c]; p = fa[p])ch[p][c] = lst;
		if(!p)return fa[lst] = 1, void();
		rg int q = ch[p][c];
		if(len[q] == len[p]+1)return fa[lst] = q, void();
		len[++siz] = len[p]+1, fa[siz] = fa[q], fa[q] = fa[lst] = siz;
		fp(i, 0, 25)ch[siz][i] = ch[q][i];
		for(; p && ch[p][c] == q; p = fa[p])ch[p][c] = siz;
	}
}sam;

struct Sam{
	int ch[maxn][26], len;
	il void build(char *s){
		len = strlen(s+1);
		fb(i, len, 1){
			fp(j, 0, 25)ch[i-1][j] = ch[i][j];
			ch[i-1][s[i]-'a'] = i;
		}
	}
}s1, s2;

struct edge{int to, ch, nxt;}e[maxn*3];
int head[maxn<<1], k;
il void add(cint &u, cint &v, cint &w){e[k] = (edge){v, w, head[u]}, head[u] = k++;}

struct node{int x, y;};
queue<node> q;

int main(){
	scanf("%s\n%s", a+1, b+1), n = strlen(a+1), m = strlen(b+1);
	fp(i, 1, m)sam.ins(b[i]-'a', i);
	s1.build(a), s2.build(b);
	
	ans = inf;
	fp(i, 1, n){
		int len = 0, nw = 1;
		while(nw && len <= n-i+1)nw = sam.ch[nw][a[i+len]-'a'], ++len;
		if(len <= n-i+1)ans = min(ans, len);
	}
	printf("%d\n", ans == inf ? -1 : ans);
	
	ans = inf;
	fp(i, 1, n){
		int len = 0, nw = 0, fl = 0;
		while((nw || !fl) && len <= n-i+1)fl = 1, nw = s2.ch[nw][a[i+len]-'a'], ++len;
		if(len <= n-i+1)ans = min(ans, len);
	}
	printf("%d\n", ans == inf ? -1 : ans);
	
	ans = inf, memset(f, -1, sizeof f), f[0][1] = 0, q.push((node){0, 1});
	while(q.size()){
		rg int x = q.front().x, y = q.front().y, nxt1, nxt2;
		q.pop();
		fp(i, 0, 25)if(nxt1 = s1.ch[x][i]){
			if(nxt2 = sam.ch[y][i]){
				if(~f[nxt1][nxt2])continue;
				f[nxt1][nxt2] = f[x][y]+1, q.push((node){nxt1, nxt2});
			}else{ans = f[x][y]+1; break;}
		}
		if(ans < inf)break;
	}
	printf("%d\n", ans == inf ? -1 : ans);
	
	while(q.size())q.pop();
	ans = inf, memset(f, -1, sizeof f), f[0][0] = 0, q.push((node){0, 0});
	while(q.size()){
		rg int x = q.front().x, y = q.front().y, nxt1, nxt2;
		q.pop();
		fp(i, 0, 25)if(nxt1 = s1.ch[x][i]){
			if(nxt2 = s2.ch[y][i]){
				if(~f[nxt1][nxt2])continue;
				f[nxt1][nxt2] = f[x][y]+1, q.push((node){nxt1, nxt2});
			}else{ans = f[x][y]+1; break;}
		}
		if(ans < inf)break;
	}
	printf("%d\n", ans == inf ? -1 : ans);
	return 0;
}
