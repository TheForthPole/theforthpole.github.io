[USACO06DEC]Milk Patterns G

2021-03-23

给定一个长度为 nnn 的串，字符集大小为 10610^6106 。要求求出最长的出现次数不小于 kkk 的子串的长度。 n,k≤20000n,k \leq 20000n,k≤20000 。

考虑建出串的 SAM\text{SAM}SAM 。那么自动机上每个节点就对应了若干个子串。这时只需要取最长的一个计算答案即可，因为它们的 endpos\text{endpos}endpos 集合一定是相同的，也就是说出现次数也是相同的。那么只需要求出每个状态的 endpos\text{endpos}endpos 集合大小（也就是这个状态的子串的出现次数）即可。
求出现次数就考虑在 parent\text{parent}parent 树上转移。初始化时令每个不是由复制产生的节点的 fff 等于 111 ，然后 dfs\text{dfs}dfs 一遍 parent\text{parent}parent 树求个子树和即可。时间复杂度 O(nlog⁡∣Σ∣)\mathcal{O} (n\log |\Sigma|)O(nlog∣Σ∣)。

#include<bits/stdc++.h>
#define rg register
#define il inline
#define cn const
#define gc getchar()
#define fp(i,a,b) for(rg int i=(a),ed=(b);i<=ed;++i)
#define go(u) for(rg int i = head[u]; ~i; i = e[i].nxt)
#define fi first
#define se second
using namespace std;
typedef cn int cint;
typedef long long LL;
il void rd(int &x){
    x = 0;
	rg int f(1); rg char c(gc);
	while(c<'0'||'9'<c){if(c=='-')f=-1;c=gc;}
	while('0'<=c&&c<='9')x=(x<<1)+(x<<3)+(c^48),c=gc;
	x*=f;
}
cint maxn = 40010;
int n, k, a[maxn], len[maxn], f[maxn], fa[maxn], siz = 1, lst = 1, ans;
map<int, int> ch[maxn];
struct edge{int to, nxt;}e[maxn*3];
int head[maxn], E;
il void add(cint &u, cint &v){e[E] = (edge){v, head[u]}, head[u] = E++;}
il void ins(cint &c, int p){
	len[++siz] = p, p = lst, lst = siz, f[siz] = 1;
	for(; p && !ch[p][c]; p = fa[p])ch[p][c] = lst;
	if(!p)return fa[lst] = 1, void();
	rg int q = ch[p][c];
	if(len[q] == len[p]+1)return fa[lst] = q, void();
	fa[++siz] = fa[q], len[siz] = len[p]+1, fa[q] = fa[lst] = siz, ch[siz] = ch[q];
	for(; p && ch[p][c] == q; p = fa[p])ch[p][c] = siz;
}
void dfs(int u){go(u)dfs(e[i].to), f[u] += f[e[i].to];}
int main(){
	rd(n), rd(k), memset(head, -1, sizeof head);
	fp(i, 1, n)rd(a[i]), ins(a[i], i);
	fp(i, 2, siz)add(fa[i], i);
	dfs(1);
	fp(i, 1, siz)if(f[i] >= k)ans = max(ans, len[i]);
	printf("%d\n", ans);
	return 0;
}
