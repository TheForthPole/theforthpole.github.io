多项式求逆

给定一个 n−1n-1n−1 次的多项式 A(x)(a0≠0)A(x)(a_0 \neq 0)A(x)(a0​​=0) ，要求一个多项式 F(x)F(x)F(x) 满足 F(x)A(x)≡1(modx)nF(x)A(x) \equiv 1 \pmod x^nF(x)A(x)≡1(modx)n 。
假设求出了 F0(x)F_0(x)F0​(x) 满足 F0(x)A(x)≡1(modx⌈n2⌉)F_0(x)A(x) \equiv 1 \pmod{x^{\lceil \frac{n}{2} \rceil}}F0​(x)A(x)≡1(modx⌈2n​⌉) ，那么有

F0(x)A(x)≡F(x)A(x)≡1(modx⌈n2⌉)A(x)(F(x)−F0(x))≡0(modx⌈n2⌉)\begin{aligned} F_0(x)A(x) \equiv F(x)A(x) \equiv 1 \pmod{x^{\lceil \frac{n}{2} \rceil}} \\A(x)(F(x)-F_0(x)) \equiv 0 \pmod{x^{\lceil \frac{n}{2} \rceil}} \end{aligned} F0​(x)A(x)≡F(x)A(x)≡1(modx⌈2n​⌉)A(x)(F(x)−F0​(x))≡0(modx⌈2n​⌉)​

由于 A(x)≠0A(x) \neq 0A(x)​=0 ，所以

F(x)−F0(x)≡0(modx⌈n2⌉)(F(x)−F0(x))2≡0(modxn)F2(x)−2F(x)F0(x)+F02(x)≡0(modxn)\begin{aligned} F(x)-F_0(x) \equiv 0 \pmod{x^{\lceil \frac{n}{2} \rceil}} \\(F(x)-F_0(x))^2 \equiv 0 \pmod{x^n} \\F^2(x)-2F(x)F_0(x)+F_0^2(x) \equiv 0 \pmod{x^n} \end{aligned} F(x)−F0​(x)≡0(modx⌈2n​⌉)(F(x)−F0​(x))2≡0(modxn)F2(x)−2F(x)F0​(x)+F02​(x)≡0(modxn)​

又因为 F(x)A(x)≡1(modxn)F(x)A(x) \equiv 1 \pmod{x^n}F(x)A(x)≡1(modxn) ，两边都乘个 A(x)A(x)A(x) ，

F(x)≡2F0(x)−A(x)F02(x)(modxn)≡F0(x)(2−A(x)F0(x))(modxn)F(x) \equiv 2F_0(x)-A(x)F_0^2(x) \pmod{x^n} \\ \equiv F_0(x)(2-A(x)F_0(x)) \pmod{x^n} F(x)≡2F0​(x)−A(x)F02​(x)(modxn)≡F0​(x)(2−A(x)F0​(x))(modxn)

所以我们可以递归求解，时间复杂度 T(n)=T(n/2)+nlog⁡n=nlog⁡nT(n) = T(n/2)+n\log n = n\log nT(n)=T(n/2)+nlogn=nlogn 。

void get_inv(int *a, int *f, int n){
	if(n == 1)return f[0] = fpow(a[0], mod-2), void();
	get_inv(a, f, n+1>>1), init(2*n-1);//init()处理数组长度和蝴蝶变换
	fp(i, 0, n-1)inv_ary[i] = a[i];
	ntt(f, 0), ntt(inv_ary, 0);
	fp(i, 0, lim-1)f[i] = 1ll*f[i]*(2-1ll*f[i]*inv_ary[i]%mod+mod)%mod;
	ntt(f, 1);
	fp(i, n, lim-1)f[i] = 0;
	fp(i, 0, lim-1)inv_ary[i] = 0;
}

多项式求 ln\rm{ln}ln 。

给定一个 n−1n-1n−1 次的多项式 A(x)(a0=1)A(x)(a_0 = 1)A(x)(a0​=1) ，要求一个多项式 F(x)F(x)F(x) 满足 F(x)≡ln⁡A(x)(modxn)F(x) \equiv \ln A(x) \pmod{x^n}F(x)≡lnA(x)(modxn) 。
两边求个导得到：

F′(x)≡A′(x)A(x)(modxn)F'(x) \equiv \frac{A'(x)}{A(x)} \pmod{x^n} F′(x)≡A(x)A′(x)​(modxn)

所以只需要对 A(x)A(x)A(x) 求个导和逆就能求出 F′(x)F'(x)F′(x) 了。最后再对 F′(x)F'(x)F′(x) 求个积分，由于 a0=1a_0=1a0​=1 ，所以常数项为 000 。时间复杂度为 O(nlog⁡n)\mathcal{O} (n\log n)O(nlogn) 。

void get_ln(int *a, int *f, int n){
	get_inv(a, ln_ary, n), init(2*n-2);
	fp(i, 1, n-1)f[i-1] = 1ll*a[i]*i%mod;
	ntt(f, 0), ntt(ln_ary, 0);
	fp(i, 0, lim-1)f[i] = 1ll*f[i]*ln_ary[i]%mod;
	ntt(f, 1);
	fp(i, n-1, lim)f[i] = 0;
	fp(i, 0, lim)ln_ary[i] = 0;
	fb(i, n-1, 1)f[i] = 1ll*f[i-1]*inv[i]%mod;
	f[0] = 0;
}

牛顿迭代

给定一个函数 G(x)G(x)G(x) ，要求一个多项式 F(x)F(x)F(x) 满足 G(F(x))≡0(modxn)G(F(x)) \equiv 0 \pmod{x^n}G(F(x))≡0(modxn) 。
假设求出了 F0(x)F_0(x)F0​(x) 满足 G(F0(x))≡0(modx⌈n2⌉)G(F_0(x)) \equiv 0 \pmod{x^{\lceil \frac{n}{2}\rceil}}G(F0​(x))≡0(modx⌈2n​⌉) ，那么把 G(x)G(x)G(x) 在 F0(x)F_0(x)F0​(x) 处泰勒展开可以得到：

G(F(x))≡G(F0(x))+G′(F0(x))(F(x)−F0(x))+G′′(F0(x))2(F(x)−F0(x))2+⋯(modxn)≡G(F0(x))+G′(F0(x))(F(x)−F0(x))≡0(modxn)F(x)≡F0(x)−G(F0(x)G′(F0(x))(modxn)G(F(x)) \equiv G(F_0(x))+G'(F_0(x))(F(x)-F_0(x))+\frac{G''(F_0(x))}{2}(F(x)-F_0(x))^2+\cdots \pmod{x^n} \\ \equiv G(F_0(x))+G'(F_0(x))(F(x)-F_0(x)) \equiv 0 \pmod{x^n} \\ F(x) \equiv F_0(x)-\frac{G(F_0(x)}{G'(F_0(x))} \pmod{x^n} G(F(x))≡G(F0​(x))+G′(F0​(x))(F(x)−F0​(x))+2G′′(F0​(x))​(F(x)−F0​(x))2+⋯(modxn)≡G(F0​(x))+G′(F0​(x))(F(x)−F0​(x))≡0(modxn)F(x)≡F0​(x)−G′(F0​(x))G(F0​(x)​(modxn)

因为 F(x)−F0(x)F(x)-F_0(x)F(x)−F0​(x) 最低次是 x⌈n2⌉x^{\lceil \frac{n}{2}\rceil}x⌈2n​⌉ ，所以从第三项开始模 xnx^nxn 就为 000 了。 注意这里的求导是求 F0(x)F_0(x)F0​(x) 的偏导 。然后就可以递归计算了。

多项式 exp\rm{exp}exp

给定一个 n−1n-1n−1 次多项式 A(x)(a0=0)A(x)(a_0=0)A(x)(a0​=0) ，要求一个多项式 F(x)F(x)F(x) 满足 F(x)≡eA(x)(modxn)F(x) \equiv e^{A(x)} \pmod{x^n}F(x)≡eA(x)(modxn) 。
两边求个 ln⁡\lnln ： ln⁡F(x)≡A(x)(modxn)\ln F(x) \equiv A(x) \pmod{x^n}lnF(x)≡A(x)(modxn) 。设 G(F(x))=ln⁡F(x)−A(x)G(F(x))=\ln F(x)-A(x)G(F(x))=lnF(x)−A(x) ，那就相当于求 F(x)F(x)F(x) 满足 G(F(x))≡0(modxn)G(F(x)) \equiv 0 \pmod{x^n}G(F(x))≡0(modxn) 。
根据牛顿迭代的式子， F(x)≡F0(x)−F0(x)(ln⁡F0(x)−A(x))≡F0(x)(1−ln⁡F0(x)+A(x))F(x) \equiv F_0(x)-F_0(x)(\ln F_0(x)-A(x)) \equiv F_0(x)(1-\ln F_0(x)+A(x))F(x)≡F0​(x)−F0​(x)(lnF0​(x)−A(x))≡F0​(x)(1−lnF0​(x)+A(x)) 。递归求解即可，时间复杂度 O(nlog⁡n)\mathcal{O} (n\log n)O(nlogn) 。

void get_exp(int *a, int *f, int n){
	if(n == 1)return f[0] = 1, void();
	get_exp(a, f, n+1>>1), get_ln(f, exp_ary, n), init(2*n-1);
	fp(i, 0, n-1)exp_ary[i] = (a[i]-exp_ary[i]+mod)%mod;
	if((++exp_ary[0]) ==  mod)exp_ary[0] = 0;
	ntt(f, 0), ntt(exp_ary, 0);
	fp(i, 0, lim-1)f[i] = 1ll*f[i]*exp_ary[i]%mod;
	ntt(f, 1);
	fp(i, n, lim-1)f[i] = 0;
	fp(i, 0, lim-1)exp_ary[i] = 0;
}

分治 FFT/NTT\rm{FFT/NTT}FFT/NTT

求出 ∏i=1n(ai,0+ai,1x+ai,2x2+⋯+ai,k+1xk+1)\prod_{i=1}^n(a_{i,0}+a_{i,1}x+a_{i,2}x^2+ \cdots +a_{i,k+1}x^{k+1})∏i=1n​(ai,0​+ai,1​x+ai,2​x2+⋯+ai,k+1​xk+1) ，要求复杂度为 O(nklog⁡2nk)\mathcal{O} (nk \log^2 nk)O(nklog2nk) 。一般情况下 kkk 都等于 111 。
显然不能从左到右乘过去，这样的话复杂度高达 O(n2)\mathcal{O} (n^2)O(n2) 。
考虑分治，每次做完左右两边的乘法，这样左右两边的长度都是 len2\frac{len}{2}2len​ 的， lenlenlen 是当前的分治区间长度。然后把两边的乘起来即可。显然复杂度降到了 O(nklog⁡2nk)\mathcal{O} (nk \log^2 nk)O(nklog2nk) 。当 k=1k=1k=1 时就是 O(nlog⁡2n)\mathcal{O} (n \log^2 n)O(nlog2n) 。
贴个 k=1k=1k=1 时的代码（a0=1a_0 = 1a0​=1）。

int div_ary[20][maxn<<2];
void divntt(int d, int l, int r){
	if(l == r)return div_ary[d][0] = 1, div_ary[d][1] = mod-a[l], void();
	int md = l+r>>1, len = r-l+2;
	divntt(d, l, md), divntt(d+1, md+1, r), init(len), ntt(div_ary[d], 0), ntt(div_ary[d+1], 0);
	fp(i, 0, lim-1)div_ary[d][i] = 1ll*div_ary[d][i]*div_ary[d+1][i]%mod;
	ntt(div_ary[d], 1);
	fp(i, len, lim-1)div_ary[d][i] = 0;
	fp(i, 0, lim-1)div_ary[d+1][i] = 0;
}
