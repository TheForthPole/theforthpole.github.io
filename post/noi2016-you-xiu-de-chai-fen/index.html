[NOI2016] 优秀的拆分

2021-03-22

如果一个字符串可以被拆分为 AABB\text{AABB}AABB 的形式，其中 A\text{A}A 和 B\text{B}B 是任意非空字符串，则我们称该字符串的这种拆分是优秀的。A,B\text{A,B}A,B 可以相同。现在给出一个长度为 nnn 的字符串 SSS ，我们需要求出，在它所有子串的所有拆分方式中，优秀拆分的总个数。这里的子串是指字符串中连续的一段。 TTT 组数据， T≤10,n≤3×104T \leq 10 , n \leq 3\times 10^4T≤10,n≤3×104 。

神仙思路题。
假设字符串以 000 开头。设 fif_ifi​ 为以 iii 为左端点的 AA\text{AA}AA 串的个数， gig_igi​ 为以 iii 为右端点的 AA\text{AA}AA 串的个数。答案显然等于

∑i=1n−1fi×gi−1\sum_{i=1}^{n-1}f_i\times g_{i-1} i=1∑n−1​fi​×gi−1​

问题是怎么求 fff 和 ggg 。
下面开始就是神仙思路了 %%%\% \% \%%%%。
首先按照长度对子串分类，计算每种长度有多少种 AA\text{AA}AA 串。考虑每隔 lenlenlen 个字符设一个关键点，求出相邻两个关键点 xxx 和 x+lenx+lenx+len 的 LCP\text{LCP}LCP 和 LCS\text{LCS}LCS 。如果 LCP+LCS>n\text{LCP+LCS} > nLCP+LCS>n ，假设 dt=LCP+LCS−ndt = \text{LCP+LCS}-ndt=LCP+LCS−n，则说明 [x−LCS,x−LCS+dt)[x-\text{LCS}, x-\text{LCS}+dt)[x−LCS,x−LCS+dt) 都能有一个长度为 lenlenlen 的 AA\text{AA}AA 串， (x+len+LCP−dt,x+len+LCP](x+len+\text{LCP}-dt, x+len+\text{LCP}](x+len+LCP−dt,x+len+LCP] 也都能有一个长度为 lenlenlen 的 AA\text{AA}AA 串。差分一下就行了。时间复杂度 O(nlog⁡n+∑i=1nni)=O(nlog⁡n)\mathcal{O} (n\log n + \sum_{i=1}^n \frac{n}{i}) = \mathcal{O} (n\log n)O(nlogn+∑i=1n​in​)=O(nlogn)。

#include<bits/stdc++.h>
#define rg register
#define il inline
#define cn const
#define gc getchar()
#define fp(i,a,b) for(rg int i=(a),ed=(b);i<=ed;++i)
#define fb(i,a,b) for(rg int i=(a),ed=(b);i>=ed;--i)
using namespace std;
typedef cn int cint;
typedef long long LL;
cint maxn = 30010;
int T, n, lg[maxn];
LL f[maxn], g[maxn], ans;
char s[maxn], t[maxn];
struct SuffixArray{
	int sa[maxn], x[maxn], y[maxn], t[maxn], len, rk[maxn], height[maxn], mn[maxn][15];
	il void get_sa(char *s){
		fp(i, 1, len = 26)t[i] = 0;
		fp(i, 1, n)++t[x[i] = s[i]-'a'+1];
		fp(i, 1, len)t[i] += t[i-1];
		fb(i, n, 1)sa[t[x[i]]--] = i;
		for(rg int k = 1; k <= n; k <<= 1){
			fp(i, 1, len)t[i] = 0;
			fp(i, 1, n)++t[x[i]];
			fp(i, 1, len)t[i] += t[i-1];
			rg int num = 0;
			fp(i, n-k+1, n)y[++num] = i;
			fp(i, 1, n)if(sa[i] > k)y[++num] = sa[i]-k;
			fb(i, n, 1)sa[t[x[y[i]]]--] = y[i], y[i] = 0;
			swap(x, y), x[sa[1]] = 1, num = 1;
			fp(i, 2, n)x[sa[i]] = (y[sa[i]] == y[sa[i-1]] && y[sa[i]+k] == y[sa[i-1]+k]) ? num : ++num;
			if(num == n)break;
			len = num;
		}
		fp(i, 1, n)y[i] = x[i] = 0;
		
		len = 0;
		fp(i, 1, n)rk[sa[i]] = i;
		fp(i, 1, n)if(rk[i] > 1){
			if(len)--len;
			rg int j = sa[rk[i]-1];
			while(i+len <= n && j+len <= n && s[i+len] == s[j+len])++len;
			height[rk[i]] = len;
		}
		
		fp(i, 2, n)mn[i][0] = height[i];
		for(rg int j = 0, s = 1; j < 14; ++j, s <<= 1)fp(i, s<<1|1, n)mn[i][j+1] = min(mn[i][j], mn[i-s][j]);
	}
	il int get_lcp(int x, int y){
		if(x == y)return n-x+1;
		x = rk[x], y = rk[y];
		if(x > y)x ^= y ^= x ^= y;
		rg int len = lg[y-x];
		return min(mn[x+(1<<len)][len], mn[y][len]);
	}
}sa1, sa2;
int main(){
	scanf("%d", &T);
	while(T--){
		scanf("%s", s+1), n = strlen(s+1), ans = 0;
		fp(i, 1, n)t[i] = s[n-i+1];
		fp(i, 2, n)lg[i] = lg[i>>1]+1;
		sa1.get_sa(s), sa2.get_sa(t);
		fp(i, 1, n)f[i] = g[i] = 0;
		fp(len, 1, n>>1){
			for(rg int i = len<<1; i <= n; i += len){
				rg int lcp = min(sa1.get_lcp(i-len, i), len);
				rg int lcs = min(sa2.get_lcp(n-i+1, n-i+1+len), len);
				if(lcp+lcs <= len)continue;
				rg int dt = lcp+lcs-len;
				++f[i-len-lcs+1], --f[i-len-lcs+1+dt], ++g[i+lcp-1], --g[i+lcp-1-dt];
			}
		}
		fp(i, 2, n)f[i] += f[i-1];
		fb(i, n-1, 1)g[i] += g[i+1];
		fp(i, 2, n)ans += 1ll*g[i-1]*f[i];
		printf("%lld\n", ans);
	}
	return 0;
}
