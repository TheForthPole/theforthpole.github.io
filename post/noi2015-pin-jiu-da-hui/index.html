给定一个由小写字母组成的长度为 nnn 的串，定义两个后缀为“ rrr 相似”的当且仅当它们长度为 rrr 的前缀相同。显然 rrr 相似能推出 r−1r-1r−1 相似。再给定每个后缀一个权值（可以为负），定义一对后缀的价值为这两个后缀的权值的乘积。现在要求你求出对于每个 r∈0,n−1r\in{0,n-1}r∈0,n−1 ，有多少对后缀是 rrr 相似的并且价值最大的一对价值为多少。n≤3×105,∣ai∣≤109n\leq 3\times 10^5,|a_i|\leq 10^9n≤3×105,∣ai​∣≤109。

设 rkirk_irki​ 为第 iii 个后缀的排名（按字典序）。
如果后缀 i,j(rki<rkj)i,j(rk_i<rk_j)i,j(rki​<rkj​) 的 LCP=rLCP=rLCP=r ，那么可以转化为 min⁡x=rki+1rkj{heightx}=r\min_{x=rk_i+1}^{rk_j} \{ height_x \} = rminx=rki​+1rkj​​{heightx​}=r。问题就变成了求有多少个区间的 heightheightheight 的最小值为 rrr 。
容易想到按 heightheightheight 从大到小加入，加入第 iii 个后缀就能合并 rki−1rk_{i-1}rki−1​ 和 rkirk_irki​ 所在的区间。所以用并查集维护一下每个点所在的区间和每个点所在区间的最大值和最小值即可。维护最小值的原因是权值可以为负，两个负权值乘起来可能更大。

#include<bits/stdc++.h>
#define rg register
#define il inline
#define cn const
#define gc getchar()
#define fp(i,a,b) for(rg int i=(a),ed=(b);i<=ed;++i)
#define fb(i,a,b) for(rg int i=(a),ed=(b);i>=ed;--i)
using namespace std;
typedef cn int cint;
typedef long long LL;
cint maxn = 300010;
cn LL inf = 0x3f3f3f3f3f3f3f3f;
int n, a[maxn];
int sa[maxn], x[maxn], y[maxn], t[maxn], len = 26, rk[maxn], hei[maxn];
int fa[maxn], siz[maxn], mx[maxn], mn[maxn];
LL sum[maxn], fmx[maxn];
char s[maxn];
struct node{int rk, h;}nd[maxn];
il bool cmp(cn node &x, cn node &y){return x.h > y.h;}
int get(int x){return fa[x] == x ? x : fa[x] = get(fa[x]);}
int main(){
	scanf("%d", &n);
	scanf("%s", s+1);
	fp(i, 1, n)scanf("%d", &a[i]);
	
	fp(i, 1, n)++t[x[i] = s[i]-'a'+1];
	fp(i, 1, len)t[i] += t[i-1];
	fb(i, n, 1)sa[t[x[i]]--] = i;
	for(rg int k = 1; k <= n; k <<= 1){
		fp(i, 1, len)t[i] = 0;
		fp(i, 1, n)++t[x[i]];
		fp(i, 1, len)t[i] += t[i-1];
		rg int num = 0;
		fp(i, n-k+1, n)y[++num] = i;
		fp(i, 1, n)if(sa[i] > k)y[++num] = sa[i]-k;
		fb(i, n, 1)sa[t[x[y[i]]]--] = y[i], y[i] = 0;
		swap(x, y);
		x[sa[1]] = 1, num = 1;
		fp(i, 2, n)x[sa[i]] = (y[sa[i]] == y[sa[i-1]] && y[sa[i]+k] == y[sa[i-1]+k]) ? num : ++num;
		if(num == n)break;
		len = num;
	}
	len = 0;
	fp(i, 1, n)rk[sa[i]] = i;
	fp(i, 1, n)if(rk[i] > 1){
		if(len)--len;
		rg int j = sa[rk[i]-1];
		while(i+len <= n && i+len <= n && s[i+len] == s[j+len])++len;
		hei[rk[i]] = len;
	}
	
	fp(i, 1, n)nd[i] = (node){i, hei[i]};
	sort(nd+2, nd+1+n, cmp);
	fp(i, 1, n)fa[i] = i, siz[i] = 1, mx[i] = mn[i] = a[i];
	fp(i, 0, n)fmx[i] = -inf;
	fp(i, 2, n){
		rg int x = get(sa[nd[i].rk]), y = get(sa[nd[i].rk-1]), h = hei[nd[i].rk];
		fa[y] = x, sum[h] += 1ll*siz[x]*siz[y], siz[x] += siz[y];
		fmx[h] = max(fmx[h], 1ll*mx[x]*mx[y]);
		fmx[h] = max(fmx[h], 1ll*mx[x]*mn[y]);
		fmx[h] = max(fmx[h], 1ll*mn[x]*mx[y]);
		fmx[h] = max(fmx[h], 1ll*mn[x]*mn[y]);
		mx[x] = max(mx[x], mx[y]), mn[x] = min(mn[x], mn[y]);
	}
	fb(i, n-2, 0)sum[i] += sum[i+1], fmx[i] = max(fmx[i], fmx[i+1]);
	fp(i, 0, n-1)printf("%lld %lld\n", sum[i], fmx[i] == -inf ? 0 : fmx[i]);
	return 0;
}
